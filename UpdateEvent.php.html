namespace Symfony\Flex\Event;

use Composer\Script\Event;

class UpdateEvent extends Event
{
    private $force;

    public function __construct(bool $force)
    {
        $this->force = $force;
    }

    public function force(): bool
    {
        return $this->force;
    }
}
